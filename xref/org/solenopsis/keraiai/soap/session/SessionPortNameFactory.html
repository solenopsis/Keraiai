17  package org.solenopsis.keraiai.soap.session;
18  
19  import javax.xml.ws.Service;
20  import org.solenopsis.keraiai.Credentials;
21  import org.solenopsis.keraiai.LoginContext;
22  
23  /**
24   * Computes a port names. For an API port (enterprise, parter, tooling, or metadata) it is the version found in credentials. For
25   * custom web services, it is the name of the web service class.
26   *
27   * @author Scot P. Floess
28   */
29  interface SessionPortNameFactory {
30      /**
31       * Creates API port names.
32       */
33      SessionPortNameFactory API_PORT_NAME_FACTORY = new ApiSessionPortNameFactory();
34  
35      /**
36       * Creates custom port names.
37       */
38      SessionPortNameFactory CUSTOM_PORT_NAME_FACTORY = new CustomSessionPortNameFactory();
39  
40      /**
41       * Computes a port name from the <code>credentials</code> and <code>service</code>. If dealing with an API webservice, the value
42       * of the version found in credentials will be used. However, for custom web services, it is the class name for the web service.
43       *
44       * @param credentials the login credentials (which contain an API version). Will be used in computing API port names.
45       * @param service     the service for whom we want a port name. Will be used when computing custom web service port names.
46       *
47       * @return the port name.
48       */
49      String computePortName(Credentials credentials, Service service);
50  
51      /**
52       * Computes the port name based upon <code>loginContext</code> and <code>service</code>. If dealing with an API webservice, the
53       * value of the version found in the login context's credentials will be used. However, for custom web services, it is the class
54       * name for the web service.
55       *
56       * @param loginContext contains the login credentials (which contain an PAI version). Will be used in compute API port names.
57       * @param service      the service for whom we want a port name. Will be used when computing custom web service port names.
58       *
59       * @return the port name.
60       */
61      String computePortName(LoginContext loginContext, Service service);
62  }
